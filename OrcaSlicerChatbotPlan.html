Overview

This document provides a comprehensive plan for implementing an AI-powered chatbot for the OrcaSlicer Calibration Tool web application. The chatbot will use OpenAI for language models and embeddings, Vercel Postgres with pgvector for storage, LightRAG for GraphRAG capabilities, and Firecrawl for documentation scraping.

Architecture Overview

  • Frontend: React component integrated into existing header
  • Backend: Vercel serverless functions with streaming responses
  • Database: Vercel Postgres with pgvector extension
  • AI/ML: OpenAI GPT-4 and embeddings
  • RAG System: LightRAG for knowledge graph management
  • Data Ingestion: Firecrawl for web scraping

Detailed Implementation Plan

LightRAG Implementation Strategy

Since LightRAG is primarily a Python implementation, we have several options:

Option 1: TypeScript Implementation (Recommended)

We’ll implement the core LightRAG concepts in TypeScript: - Entity extraction using OpenAI function calling - Relationship mapping with graph algorithms - Dual-level retrieval (low-level for specific entities, high-level for concepts) - PostgreSQL with pgvector for storage

Option 2: Python Microservice

Run LightRAG as a separate Python service and communicate via API: - Pros: Use official LightRAG implementation - Cons: Additional infrastructure complexity

Option 3: Hybrid Approach

Use TypeScript for main logic with Python scripts for batch processing: - TypeScript API for real-time queries - Python scripts for document ingestion and graph building

Decision: We’ll go with Option 1 (TypeScript Implementation) for better integration with our Vercel stack.

CREATE TABLE IF NOT EXISTS kg_entities (
  id SERIAL PRIMARY KEY,
  name TEXT NOT NULL UNIQUE,
  entity_type VARCHAR(100),
  description TEXT,
  embedding vector(1536),

  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Knowledge graph relationships
CREATE TABLE IF NOT EXISTS kg_relationships (
  id SERIAL PRIMARY KEY,
  source_entity_id INTEGER REFERENCES kg_entities(id),
  target_entity_id INTEGER REFERENCES kg_entities(id),
  relationship_type VARCHAR(100),
  weight FLOAT DEFAULT 1.0,
  metadata JSONB DEFAULT '{}',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

LightRAG Entity Extraction

// src/lib/lightrag/entityExtractor.ts
import { OpenAI } from 'openai';

export class EntityExtractor {
  private openai: OpenAI;

  constructor() {
    this.openai = new OpenAI({
      apiKey: process.env.OPENAI_API_KEY
    });
  }

  async extractEntities(text: string): Promise<Entity[]> {
    const completion = await this.openai.chat.completions.create({
      model: "gpt-4-turbo-preview",
      messages: [
        {
          role: "system",
          content: `Extract entities related to 3D printing from the text.
          Categories: tool, material, setting, problem, solution, component, process.
          Return JSON array of entities with name, type, and description.`
        },
        {
          role: "user",
          content: text
        }
      ],
      response_format: { type: "json_object" },
      temperature: 0.1
    });

    const result = JSON.parse(completion.choices[0].message.content || '{"entities": []}');
    return result.entities;
  }

  async extractRelationships(entities: Entity[], context: string): Promise<Relationship[]> {
    // Extract relationships between entities
    const entityNames = entities.map(e => e.name).join(', ');
    
    const completion = await this.openai.chat.completions.create({
      model: "gpt-4-turbo-preview",
      messages: [
        {
          role: "system",
          content: `Given these entities: ${entityNames}
          Extract relationships between them from the context.
          Relationship types: requires, solves, causes, prevents, improves, relates_to.
          Return JSON array of relationships.`
        },
        {
          role: "user",
          content: context
        }
      ],
      response_format: { type: "json_object" },
      temperature: 0.1
    });

    const result = JSON.parse(completion.choices[0].message.content || '{"relationships": []}');
    return result.relationships;
  }
}

Dual-Level Retrieval Implementation

// src/lib/lightrag/retrieval.ts
export class DualLevelRetrieval {
  async lowLevelRetrieve(query: string, topK: number = 5) {
    // Direct entity search
    const queryEmbedding = await this.generateEmbedding(query);
    
    const sql = `
      SELECT e.*, 
        1 - (e.embedding <=> $1::vector) as similarity
      FROM kg_entities e
      WHERE 1 - (e.embedding <=> $1::vector) > 0.7
      ORDER BY similarity DESC
      LIMIT $2
    `;
    
    return await db.query(sql, [queryEmbedding, topK]);
  }

  async highLevelRetrieve(query: string, topK: number = 5) {
    // Conceptual search with graph traversal
    const concepts = await this.extractConcepts(query);
    
    // Find entities related to concepts
    const sql = `
      WITH RECURSIVE related_entities AS (
        -- Start with entities matching concepts
        SELECT e.id, e.name, e.embedding, 0 as depth
        FROM kg_entities e
        WHERE e.name ILIKE ANY($1::text[])
        
        UNION
        
        -- Traverse relationships
        SELECT e.id, e.name, e.embedding, re.depth + 1
        FROM kg_entities e
        JOIN kg_relationships r ON e.id = r.target_entity_id
        JOIN related_entities re ON r.source_entity_id = re.id
        WHERE re.depth < 2
      )
      SELECT DISTINCT * FROM related_entities
      ORDER BY depth, name
      LIMIT $2
    `;
    
    return await db.query(sql, [concepts, topK]);
  }
}

Chat API with LightRAG Integration

// src/app/api/chat/route.ts
import { DualLevelRetrieval } from '@/lib/lightrag/retrieval';
import { streamText } from 'ai';
import { openai } from '@ai-sdk/openai';

export async function POST(req: Request) {
  const { messages, sessionId, files } = await req.json();
  const retrieval = new DualLevelRetrieval();
  
  // Get user's query
  const lastMessage = messages[messages.length - 1].content;
  
  // Perform dual-level retrieval
  const [lowLevel, highLevel] = await Promise.all([
    retrieval.lowLevelRetrieve(lastMessage),
    retrieval.highLevelRetrieve(lastMessage)
  ]);
  
  // Process uploaded files if any
  let fileContext = '';
  if (files && files.length > 0) {
    fileContext = await processUploadedFiles(files);
  }
  
  // Build context
  const context = `
    Direct Information:
    ${lowLevel.map(e => `- ${e.name}: ${e.description}`).join('\n')}
    
    Related Concepts:
    ${highLevel.map(e => `- ${e.name}: ${e.description}`).join('\n')}
    
    ${fileContext ? `User Files:\n${fileContext}` : ''}
  `;
  
  // Stream response
  const result = await streamText({
    model: openai('gpt-4-turbo-preview'),
    messages: [
      {
        role: 'system',
        content: `You are an expert OrcaSlicer assistant. Use the provided context to answer questions about 3D printing calibration, troubleshooting, and settings.
        
        Context:
        ${context}`
      },
      ...messages
    ],
    temperature: 0.3,
    maxTokens: 2000,
  });
  
  return result.toAIStreamResponse();
}
